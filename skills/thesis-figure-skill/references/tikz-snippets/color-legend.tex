% =====================================================================
% Snippet: color-legend.tex
% =====================================================================
% USAGE: Horizontal color legend strip — color square + label pairs.
% Used at the bottom of complete figures (see examples 06 Transformer).
%
% KEY DESIGN: Uses anchor-relative chaining (each item placed at
% previous.east + offset) so labels CANNOT overlap the next swatch,
% regardless of label length. v1 used fixed itemGap which broke for
% labels longer than gap (e.g., "Multi-Head Attention" 3cm > 2.6cm gap).
%
% Agent / sub-agent customization: edit \legdata pairs (color/label).
% Number of items 4-8 recommended. Hard-coded 6 items below as
% \foreach inside named-coordinate chains is unreliable in TikZ.
% =====================================================================

\documentclass[tikz,border=10pt]{standalone}
\usepackage{xcolor,tikz}
\usetikzlibrary{positioning,calc}

\definecolor{acaBlueLine}{HTML}{3060A8}
\definecolor{acaPurpleLine}{HTML}{6E3FA0}
\definecolor{acaOrangeLine}{HTML}{D06020}
\definecolor{acaGreenLine}{HTML}{389050}
\definecolor{acaTealLine}{HTML}{1F8F8F}
\definecolor{acaGoldLine}{HTML}{B08820}
\definecolor{acaGreyLine}{HTML}{606060}

\tikzset{
  legend swatch/.style={
    fill=#1, draw=#1, line width=0.4pt,
    minimum width=0.25cm, minimum height=0.25cm,
    inner sep=0pt, anchor=west,
  },
  legend text/.style={
    anchor=west, font=\scriptsize, color=acaGreyLine,
    inner sep=2pt,
  },
}

\begin{document}
\begin{tikzpicture}

%% COPY-START =========================================================
% Each item = swatch (named sq\i) + label (named lab\i) placed at
% PREVIOUS label.east + offset. Self-aligning, zero overlap guarantee.

% --- Label "Color Legend:" anchor ---
\node[font=\scriptsize\bfseries, color=acaGreyLine, anchor=west]
  (lab0) at (0, 0) {Color Legend:};

% --- Item 1: Multi-Head Attention (blue) ---
