% =====================================================================
% Snippet: summary-bar.tex
% =====================================================================
% USAGE: Bottom pipeline summary bar — stages connected by labeled
% arrows. e.g.: Molecule -[GAT]-> (A,X) -[GAT]-> α_ij -[Multi-head]-> ...
%
% KEY DESIGN: Uses anchor-relative chaining (each stage placed at
% previous.east + fixed arrow distance) so long stage labels
% (e.g., "Atom Importance") cannot overlap previous stage's edge label.
% v1 used fixed itemGap which broke for long labels.
%
% Sub-agent customization: edit the 5 \node...at calls + 4 \draw...node
% calls below. Stage count = 5 hard-coded; for different counts, add/
% remove (node + draw) pairs.
% =====================================================================

\documentclass[tikz,border=10pt]{standalone}
\usepackage{xcolor,tikz,amsmath,amssymb}
\usetikzlibrary{positioning,calc,arrows.meta,bending}

\definecolor{acaGreyLine}{HTML}{606060}
\definecolor{acaGreyFill}{HTML}{F0F0F0}
\definecolor{acaPurpleLine}{HTML}{6E3FA0}

\tikzset{
  sum stage/.style={
    rectangle, rounded corners=3pt,
    fill=acaGreyFill, draw=acaGreyLine, line width=0.5pt,
    font=\scriptsize, text=acaGreyLine,
    inner sep=4pt, minimum height=0.5cm,
  },
  sum arrow/.style={
    -{Stealth[length=4pt 1.0, width'=0pt 0.5]},
    line width=0.6pt, color=acaPurpleLine,
    shorten >=1pt, shorten <=1pt,
  },
  edge label/.style={
    midway, above, font=\tiny, color=acaPurpleLine,
    inner sep=1pt, yshift=-1pt,
  },
}

\begin{document}
\begin{tikzpicture}

%% COPY-START =========================================================
% Stages chained via right=0.7cm of <prev>. Arrow gap fixed regardless
% of stage label length — guaranteed no overlap with edge label above.

\def\arrgap{1.6cm}   % fixed gap between adjacent stages — must be
                      % wider than longest edge label (e.g.,
                      % "Readout+MLP" ≈ 1.3cm in \tiny). If your edge
                      % labels are longer, increase to 2.0cm+.

% Stage 1 (anchor at origin)
\node[sum stage] (sb1) at (0, 0) {Molecule};

% Stage 2 (placed right of sb1, with arrow between)
\node[sum stage, right=\arrgap of sb1] (sb2) {$(\mathbf{A}, \mathbf{X})$};
\draw[sum arrow] (sb1.east) -- (sb2.west)
  node[edge label] {GAT};

% Stage 3
\node[sum stage, right=\arrgap of sb2] (sb3) {$\alpha_{ij}$ Weighted};
\draw[sum arrow] (sb2.east) -- (sb3.west)
  node[edge label] {GAT};

% Stage 4
\node[sum stage, right=\arrgap of sb3] (sb4) {128-dim};
\draw[sum arrow] (sb3.east) -- (sb4.west)
  node[edge label] {Multi-head};

% Stage 5
\node[sum stage, right=\arrgap of sb4] (sb5) {Atom Importance};
\draw[sum arrow] (sb4.east) -- (sb5.west)
  node[edge label] {Readout+MLP};

%% COPY-END ===========================================================

\end{tikzpicture}
\end{document}
